\begin{slide}

 \itemtrig\ n processadores desejam escrever simultaneamente (com prio\-ridade) em uma variável $A$, valores $B[1], B[2], \cdots, B[n]$, respectivamente.
 
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW com prioridade $\rightarrow$ CRCW Forte}
 
\itemtrig\  Antes da escrita simultânea, os processadores escrevem simultaneamente (forte) em uma variável x, o seu número de prioridade.
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW com prioridade $\rightarrow$ CRCW Forte}

\itemtrig\  Como resultado, o valor de x será o número de prioridade maior.

\vspace{-1cm} \itemtrig\  O processador, cuja prioridade é igual a x, escreve então, exclusivamente, em $A$, o elemento de B.
  
\end{slide}
\begin{slide}
  \stitle{Simulação CRCW com prioridade $\rightarrow$ CRCW Forte}

Algoritmo para CRCW com prioridade

\begin{codebox}
  \zi \For $1 \leq i \leq n$ \DoPar 
  \zi     $A := B[i]$
  \zi \End   
\end{codebox}

Algoritmo para CRCW forte
\begin{codebox}
  \zi \For $1 \leq i \leq n$ \DoPar 
  \zi     $x := i$	
  \zi \If $x = i$ \Then
  \zi     $A := B[i]$
  \zi \End  
  \zi \End   
\end{codebox}
  
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW Forte $\rightarrow$ CRCW Fraco}
  
  Temos um algoritmo para CRCW forte e uma máquina CRCW fraco.
  
  \itemtrig\ n processadores desejam escrever simultaneamente em uma variável $A$, valores  $B[1], B[2], \cdots, B[n]$, respectivamente. 
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW Forte $\rightarrow$ CRCW Fraco}
Para simular esta escrita simultânea (forte) inicializamos um vetor $F$ de flags de n posições com valor 1.

Utilizamos $O(n^2)$ processadores para comparar todos os possíveis pares de elementos de $B$. Para cada para $(i,j)$ comparado $i<j$, colocamos 0 na flag correspondente ao elemento de $B$ que é o menor dos dois. Caso os dois elementos sejam iguais, colocamos 0 na flag correspondente ao elemento de $B$ que tem maior índice dos dois.
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW Forte $\rightarrow$ CRCW Fraco}
  Como resultado, apenas uma flag de $F$ ficará com 1, e esta é a flag correspondente a um dos elementos de $B$ de maior valor.
  
  O processador correspondente a este elemento escreve exclusivamente em $A$, o seu elemento de $B$.
\end{slide}

\begin{slide}
  \stitle{Simulação CRCW Forte $\rightarrow$ CRCW Fraco}

Algoritmo para CRCW forte

\begin{codebox}
  \zi \For $1 \leq i \leq n$ \DoPar 
  \zi     $A := B[i]$
  \zi \End   
\end{codebox}
\end{slide}

\begin{slide}
Algoritmo para CRCW fraco

\begin{codebox}
  \zi \For $1 \leq i \leq n$ \DoPar 
  \zi     $F[i] := 1$	
  \zi \End    
  \zi \For $1 \leq i,j \leq n$, $i<j$ \DoPar 
  \zi \If $B[i] < B[j]$ \Then
  \zi     $F[i] := 0$
  \zi \ElseIf $B[i] > B[j]$ \Then
  \zi     $F[j] := 0$
  \zi \Else  
  \zi     $F[j] := 0$
  \zi \End  
  \zi \End  
\end{codebox}
\end{slide}

\begin{slide}
continuação...
\begin{codebox}  
  \zi \For $1 \leq i \leq n$ \DoPar 
  \zi \If $F[i] = 1$ \Then
  \zi     $A := B[i]$
  \zi \End  
  \zi \End     
\end{codebox}

\end{slide}
