\begin{slide}
  \stitle{Pointer Jumping}
Descrição da técnica

 \itemtrig\ Técnica normalmente aplicada sobre uma lista encadeada de elementos.
 
  \itemtrig\ Para cada elemento da lista é associado um processador.
  \end{slide}
 
 
 
\begin{slide}
  \stitle{Pointer Jumping}
 \itemtrig\ A técnica de pointer jumping consiste de atualizar o sucessor (ou seguinte) de cada vértice com o sucessor do sucessor.
\end{slide}
  
\begin{slide}
  \stitle{Pointer Jumping}
  
   \itemtrig\ Em um determinado passo do algoritmo, resolvemos o pro\-blema para todos os elementos da lista que estão até uma certa distância de um elemento individual da lista. 

   \itemtrig\ Esta distância dobra a cada passo. Logo, depois de $k$ passos, o problema foi resolvido para todos os elementos da lista que estão até uma distância $2^k$.
   \end{slide}
  
\begin{slide}
  \stitle{Exemplo: List Ranking}
Distância dos elementos da lista ao fim da lista.

   \itemtrig\ Dada uma lista encadeada de $n$ elementos, determinar a distância de cada elemento da lista ao fim da lista. 
   
   Idéia:
   
   \itemtrig\ Associar um processador para cada elemento da lista
   
   \itemtrig\ A cada passo, o processador responsável pelo elemento $i$ duplica sua estimativa da distância até o fim da lista.  
   \end{slide}
  
\begin{slide}
  \stitle{Exemplo: List Ranking}
  Entrada:

\vspace{-1cm} \itemtrig\   $n$: número de elementos da lista
   
\vspace{-1cm} \itemtrig\ $L$ : lista encadeada

\vspace{-1cm} \itemtrig\ $prox[i]$ : elemento de $L$ seguinte ao elemento $i$. Se $i$ é o último, então $prox[i] = nil$.

 
  Estrutura auxiliar:

\vspace{-1cm} \itemtrig\ $p[i]$ : inicialmente terá cópia de $prox[i]$
   
  Saída:
  
\vspace{-1cm} \itemtrig\ $dist[i]$ : distância do elemento $i$ ao fim de $L$.         
     
\end{slide}

 \begin{slide}
  \stitle{List Ranking}
\textbf{\blue Algoritmo}
\vspace{-1cm} \begin{codebox}
  \zi \For $i \in L$ \DoPar 
  \zi     $p[i] := prox[i]$
  \zi \If $p[i] = nil$ \Then
  \zi $dist[i] := 0$
  \zi \Else
  \zi $dist[i] := 1$
  \zi \End   
  \zi \End   
\end{codebox}
\end{slide}

 \begin{slide}
  \stitle{List Ranking}
continuação...
\begin{codebox}
  \zi \For $j := 1$ \To $\lceil \log_2 n \rceil$ \DoSeq 
  \zi \For $i \in L$ \DoPar 
  \zi \If $p[i] \neq nil$ \Then
  \zi $dist[i] := dist[i] + dist[p[i]]$
  \zi $p[i]:= p[p[i]]$  
  \zi \End   
  \zi \End   
\end{codebox}

\end{slide}

\begin{slide}
 \stitle{Construção de Sub-listas}  
Ideia:

  \vspace{-1cm}   \hspace{1cm} \itemtrig\ Aplicar duplicação recursiva, pulando apenas elementos não marcados. 
  
  \vspace{-1cm}   \hspace{1cm} \itemtrig\ Ao final deste passo, a lista só possui elementos marcados, com a possível exceção do primeiro elemento. 
  
  \vspace{-1cm}   \hspace{1cm} \itemtrig\ Se necessário então, corrige o primeiro elemento da sub-lista.

\begin{slide}

 Entrada: 
 
  \vspace{-1cm}   \hspace{1cm} \itemtrig\ $n$ : o número de elementos da lista

  \vspace{-1cm}   \hspace{1cm} \itemtrig\ $inicio$ : ponteiro para o primeiro elemento da lista.
    
  \vspace{-1cm}   \hspace{1cm} \itemtrig\ $prox[i]$ : ponteiro para o elemento seguinte ao elemento $i$ em $L$. Se $i$ é o último elemento de $L$, $prox[i] = nil$

\begin{slide}
 \stitle{Determinar raízes de uma floresta} 
Descrição: 

   \vspace{-1cm}   \hspace{1cm} \itemtrig\ Dada uma floresta $F$ de árvores enraizadas, deseja-se determinar, para cada vértice $i$ de $F$, a raiz da árvore que contém $i$.
 
\end{slide}

\begin{slide}
 \stitle{Determinar raízes de uma floresta} 
Entrada:
 
   \vspace{-1cm}   \hspace{1cm} \itemtrig\ $n$ : número de vértices de $F$.
    
   \vspace{-1cm}   \hspace{1cm} \itemtrig\ $pai[i]$ : vértice pai do vértice $i$ em $F$. Se $i$ é uma raiz, $pai[i] = -1$

Saída:

   \vspace{-1cm}   \hspace{1cm} \itemtrig\ $raiz[i]$ : vértice raiz da árvore que contém o vértice $i$.
\end{slide}

\begin{slide}
 \stitle{Determinar raízes de uma floresta} 
 Idéia:
 
   \vspace{-1cm}   \hspace{1cm} \itemtrig\ Aplicar duplicação recursiva sobre o pai. Inicialmente cada vértice sabe o seu pai. Depois de um passo, cada vértice sabe o pai do seu pai, e assim por diante, até chegar a raiz.
\end{slide}

\begin{slide}
 \stitle{Determinar raízes de uma floresta} 
\textbf{\blue Algoritmo}
\vspace{-1cm}\begin{codebox}
  \zi \For $0 \leq i \leq n - 1$ \DoPar 
  \zi $raiz[i]:= pai[i]$  
  \zi \If $raiz[i] \neq -1$ \Then
  \zi $raiz[i] := i$
  \zi \End   
  \zi \End 
  \zi \Comment Duplicação recursiva  
  \zi \For $0 \leq i \leq n - 1$ \DoPar 
  \zi \While $raiz[i] \neq raiz[raiz[i]]$ \DoSeq
  \zi $raiz[i] := raiz[raiz[i]]$  
  \zi \End   
\end{codebox}
\end{slide}


\begin{slide}
 \stitle{Outra maneira de escrever o algoritmo} 
\textbf{\blue Algoritmo}
\vspace{-1cm}\begin{codebox}
  \zi \For $0 \leq i \leq n - 1$ \DoPar

Saída:
    
    \vspace{-1cm}   \hspace{1cm} \itemtrig\ $soma[i]$: soma dos pesos dos vértices no caminho do vértice $i$ até a raiz de sua árvore.

\begin{slide}
 \stitle{Soma de Prefixos em uma Floresta}   
 Idéia:
 
     \vspace{-0.8cm}   \hspace{1cm} \itemtrig\ Aplicar duplicação recursiva sobre o pai. Inicialmente cada vértice sabe o seu peso. Depois de um passo, cada vértice sabe a soma do seu peso com o peso do seu pai, e assim por diante, até chegar a raiz.
 
\end{slide}

\begin{slide}
 \stitle{Soma de Prefixos em uma Floresta}     
\textbf{\blue Algoritmo}
\vspace{-1cm}\begin{codebox}
  \zi \For $0 \leq i \leq n - 1$ \DoPar 
  \zi $p[i]:= pai[i]$  
  \zi $soma[i]:= peso[i]$  
  \zi \End 
  \zi \For $0 \leq i \leq n - 1$ \DoPar 
  \zi \While $p[i] \neq nil$ \DoSeq
  \zi $soma[i] := soma[i] + soma[p[i]]$  
  \zi $p[i] := p[p[i]]$
  \zi \End   
  \zi \End   
\end{codebox}
\end{slide}
